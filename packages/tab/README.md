## Install

```sh
$ npm install @salte-auth/tab
```

## Usage

```js
import { SalteAuth } from '@salte-auth/salte-auth';
import { GitHub } from '@salte-auth/github';
import { Tab } from '@salte-auth/tab';

const auth = new SalteAuth({
  providers: [
    new GitHub({
      clientID: '12345'
    })
  ],

  handlers: [
    new Tab({
      default: true
    })
  ]
});

auth.login('github');
```
